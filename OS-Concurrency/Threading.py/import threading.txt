import threading
import time

def worker(name):
    for i in range(3):
        print(f"{name} is working... {i}")
        time.sleep(1)

# Create threads
t1 = threading.Thread(target=worker, args=("Thread 1",))
t2 = threading.Thread(target=worker, args=("Thread 2",))

# Start threads
t1.start()
t2.start()

# Wait for both threads to finish
t1.join()
t2.join()

print("Threading demo completed!")
